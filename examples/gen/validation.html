Example: Validation

Use Google's libphonenumber to validate the telephone number on the change event.

Markup

<script src="../../lib/libphonenumber/isValidIntlNumber.js"></script>

<input id="phone" type="tel" placeholder="e.g. +1 702 123 4567">

Code

var telInput = $("#phone");

// initialise plugin
telInput.intlTelInput();

// on change: check for errors
telInput.change(function() {
  if (isValidIntlNumber($(this).val())) {
    $(this).removeClass("error");
  } else {
    $(this).addClass("error");
  }
});

// on keyup remove any error
telInput.keyup(function() {
  $(this).removeClass("error");
});
